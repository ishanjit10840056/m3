pods ->

mkdir /k8s-code

cd /k8s-code
vim mydeployment.yaml

apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx-deployment
  labels:
    app: nginx
spec:
  replicas: 3
  selector:
    matchLabels:
      app: nginx
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
      - name: nginx
        image: nginx:latest
        ports:
        - containerPort: 80

save and exit	
kubectl apply -f mydeployment.yaml
kubectl get deployment
deployment will be live


do this:
kubectl expose deployment nginx-deployment --port=80 --target-port --type=LoadBalancer


service code:

apiVersion: v1
kind: Service
metadata:
  name: nginx-service
spec:
  selector:
    app: nginx
  ports:
    - protocol: TCP
      port: 80
      targetPort: 80
  type: LoadBalancer

kubectl apply -f service.yaml
